// ─── Utility ────────────────────────────────────────────────────────────────
function clsx(...args) {
  return args.filter(Boolean).join(" ");
}

// ─── Icons (inline SVG) ─────────────────────────────────────────────────────
const Icon = ({ d, d2, type = "path", ...props }) => (
  <svg
    viewBox="0 0 24 24"
    fill="none"
    stroke="currentColor"
    strokeWidth={2}
    strokeLinecap="round"
    strokeLinejoin="round"
    {...props}
  >
    {type === "path" && <path d={d} />}
    {d2 && <path d={d2} />}
  </svg>
);

function Pill({ children, color = "green" }) {
  const map = {
    green: { bg: "#DCFCE7", color: "#14532D" },
    amber: { bg: "#FEF3C7", color: "#78350F" },
    red:   { bg: "#FEE2E2", color: "#7F1D1D" },
    blue:  { bg: "#DBEAFE", color: "#1E3A8A" },
    gray:  { bg: "#F1F5F9", color: "#334155" },
  };
  const s = map[color] || map.gray;
  return (
    <span
      style={{
        display: "inline-flex",
        alignItems: "center",
        padding: "2px 9px",
        borderRadius: 20,
        fontSize: 11,
        fontWeight: 500,
        background: s.bg,
        color: s.color,
        whiteSpace: "nowrap",
      }}
    >
      {children}
    </span>
  );
}

// ─── Attendance Chart ─────────────────────────────────────────────────────────
function AttendanceChart() {
  const canvasRef = useRef(null);
  const chartRef = useRef(null);

  useEffect(() => {
    let Chart;
    const script = document.createElement("script");
    script.src = "https://cdnjs.cloudflare.com/ajax/libs/Chart.js/4.4.1/chart.umd.js";
    script.onload = () => {
      Chart = window.Chart;
      if (!canvasRef.current) return;
      if (chartRef.current) chartRef.current.destroy();
      chartRef.current = new Chart(canvasRef.current, {
        type: "bar",
        data: {
          labels: ["Monday", "Tuesday", "Wednesday", "Thursday", "Friday"],
          datasets: [
            {
              label: "Present",
              data: [88, 82, 91, 87, 85],

              categoryPercentage: 0.7,

              data: [39, 45, 36, 40, 42],
              backgroundColor: "#E2E4EA",
              borderRadius: 5,
              barPercentage: 0.55,
              categoryPercentage: 0.7,
            },
          ],
        },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          plugins: {
            legend: { display: false },
            tooltip: {
              callbacks: {
                label: (ctx) => ` ${ctx.dataset.label}: ${ctx.parsed.y} workers`,

            },
          },
          scales: {
            x: {
              grid: { display: false },
              ticks: { font: { family: "'DM Sans', sans-serif", size: 11 }, color: "#8B90A0" },

    document.head.appendChild(script);

// ─── Clock Hook ───────────────────────────────────────────────────────────────
function useClock() {
  const [time, setTime] = useState(() =>
    new Date().toLocaleTimeString("en-GB", { hour: "2-digit", minute: "2-digit", second: "2-digit" })
  );
  useEffect(() => {
    const id = setInterval(
      () =>
        setTime(
          new Date().toLocaleTimeString("en-GB", { hour: "2-digit", minute: "2-digit", second: "2-digit" })
        ),
      1000
    );
    return () => clearInterval(id);
  }, []);
  return time;
}
